%%% ============================================
%%% Безопасная настройка шрифтов и темы
%%% ============================================

%%% Не загружаем plex-otf вообще — он ломает математику в XeLaTeX
%%% Вместо этого используем стандартные шрифты Latin Modern,
%%% которые гарантированно поддерживают кириллицу и математику

\usepackage{lmodern}
\usepackage{amssymb,amsmath}

%%% Настройка основного шрифта с засечками (как IBM Plex Serif, но работающий)
\setmainfont{DejaVu Serif}

%%% Настройка шрифта без засечек для заголовков (как IBM Plex Sans)
\setsansfont{DejaVu Sans}

%%% Настройка моноширинного шрифта (как IBM Plex Mono)
\setmonofont{DejaVu Sans Mono}

%%% Сообщаем beamer использовать эти шрифты
\usefonttheme{professionalfonts}

%%% Настройка математических шрифтов
\setmathrm{DejaVu Serif}
\setmathsf{DejaVu Sans}
\setmathtt{DejaVu Sans Mono}

%%% Load theme
\IfFileExists{beamerthemegotham.sty}{
  \usetheme{gotham}
  \gothamset{
    numbering=totalpagenumber,
    parttocframe default=off,
    sectiontocframe default=off,
    subsectiontocframe default=off,
  }
}{
  \usetheme{Madrid}
}
